## Simple Text Generation

```typescript
import { createOpenAI } from '@dysporium-sdk/openai';
import { generateText } from '@dysporium-sdk/core';

const openai = createOpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

const result = await generateText({
  model: openai('gpt-5.1'),
  prompt: 'Write a haiku about TypeScript',
});

console.log(result.text);
```

## Conversational Chat

```typescript
const messages = [
  { role: 'system', content: 'You are a helpful assistant.' },
  { role: 'user', content: 'What is TypeScript?' },
];

const result = await generateText({
  model: openai('gpt-5.1'),
  messages,
});

console.log(result.text);
```

## Streaming Response

```typescript
import { streamText } from '@dysporium-sdk/core';

await streamText({
  model: openai('gpt-5.1'),
  prompt: 'Tell a story',
  onChunk: (chunk) => {
    if (chunk.type === 'text-delta') {
      process.stdout.write(chunk.textDelta);
    }
  },
});
```

## Using the Client

```typescript
import { DysporiumClient } from '@dysporium-sdk/core';

const client = new DysporiumClient(openai('gpt-5.1'));

const result = await client.generate({
  prompt: 'Hello, world!',
});

console.log(result.text);
```

## Advanced Parameters

Control generation with advanced parameters:

```typescript
const result = await generateText({
  model: openai('gpt-5.1'),
  prompt: 'Write a creative story',
  temperature: 0.8,
  frequencyPenalty: 0.5,  // Reduce repetition
  presencePenalty: 0.3,   // Encourage new topics
  seed: 42,               // Reproducible outputs
  maxTokens: 1000,
});

console.log(result.text);
```

## Anthropic Extended Thinking

Use extended thinking with Claude 3.5+ models:

```typescript
import { createAnthropic } from '@dysporium-sdk/anthropic';

const anthropic = createAnthropic({
  apiKey: process.env.ANTHROPIC_API_KEY,
});

const result = await generateText({
  model: anthropic('claude-3-5-sonnet-20241022'),
  prompt: 'Solve this complex problem',
  thinking: {
    type: 'enabled',
    budgetTokens: 10000,
  },
});
```
